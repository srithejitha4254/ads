#include <stdio.h>
#define MAX 5

int stack[MAX], top = -1;

void push(int x) {
    if (top == MAX - 1) return;
    stack[++top] = x;
}

void pop() {
    if (top == -1) return;
    top--;
}

void display() {
    for (int i = top; i >= 0; i--)
        printf("%d ", stack[i]);
}

int main() {
    push(10);
    push(20);
    push(30);
    pop();
    display();
    return 0;
}
