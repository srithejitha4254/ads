#include <stdio.h>

int main() {
    int n, num = 4, count = 0, i, flag;

    printf("Enter n value: ");
    scanf("%d", &n);

    while (count < n) {
        flag = 0;
        for (i = 2; i <= num / 2; i++) {
            if (num % i == 0) {
                flag = 1;
                break;
            }
        }
        if (flag == 1) {
            printf("%d ", num);
            count++;
        }
        num++;
    }

    return 0;
}
