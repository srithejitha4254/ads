#include <stdio.h>

int main() {
    int a[5] = {2, 4, 6, 8, 10};
    int key = 8, low = 0, high = 4, mid;

    while (low <= high) {
        mid = (low + high) / 2;
        if (a[mid] == key) {
            printf("Found at %d", mid);
            return 0;
        }
        if (a[mid] < key) low = mid + 1;
        else high = mid - 1;
    }
    printf("Not found");
    return 0;
}
