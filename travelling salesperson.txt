#include <stdio.h>
#include <limits.h>
#define MAX 20
int n;
int graph[MAX][MAX];
int bestCost = INT_MAX;
int bestPath[MAX];
int path[MAX];
int visited[MAX];
int bound(int level, int cost) {
    int est = cost;
    for (int i = level; i < n; i++) {
        int minEdge = INT_MAX;
        for (int j = 0; j < n; j++) {
            if (i != j && graph[i][j] < minEdge)
                minEdge = graph[i][j];
        }
        est += minEdge;
    }
    return est;
}
void tsp(int level, int cost) {
    if (level == n) {
        cost += graph[path[n-1]][path[0]];
        if (cost < bestCost) {
            bestCost = cost;
            for (int i = 0; i < n; i++)
                bestPath[i] = path[i];
        }
        return;
    }

    for (int i = 1; i < n; i++) {
        if (!visited[i]) {
            int newCost = cost + graph[path[level-1]][i];
            int est = bound(level, newCost);
            if (est < bestCost) {
                path[level] = i;
                visited[i] = 1;
                tsp(level + 1, newCost);
                visited[i] = 0;
            }
        }
    }
}

int main() {
    printf("Enter number of cities: ");
    scanf("%d", &n);

    printf("Enter cost adjacency matrix:\n");
    for (int i = 0; i < n; i++)
        for (int j = 0; j < n; j++)
            scanf("%d", &graph[i][j]);
    for (int i = 0; i < n; i++)
        visited[i] = 0;
    path[0] = 0;
    visited[0] = 1;
    tsp(1, 0);
    printf("Minimum Tour Cost: %d\n", bestCost);
    printf("Optimal Path: ");
    for (int i = 0; i < n; i++)
        printf("%d -> ", bestPath[i]);
    printf("%d\n", bestPath[0]);
    return 0;
}
